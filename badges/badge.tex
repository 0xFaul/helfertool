%
% Badge
%

% badge width
\newlength{\badgewidth}
\setlength{\badgewidth}{8cm}

% front
\newcommand*\badgefront[1][]{%
    \begingroup%
    \setkeys{badge}{#1}%
        \begin{tikzpicture}[x=1cm,y=1cm, inner sep=0, outer sep=0, anchor=north west]
            % background
            \node (bg) at (0,0) {\includegraphics[width=\badgewidth]{bg.png}};

            % all coordinates are relative to background (0,0) to (1,1)
            \begin{scope}[x={(bg.south east)},y={(bg.north west)}]
                % name
                \node[draw=none] at (0.1,0.1) {\badge@prename\ \badge@surname};
                % job
                \node[draw=none] at (0.2,0.2) {\badge@job};
            \end{scope}
        \end{tikzpicture}%
    \endgroup%
}

%back
\newcommand*\badgeback[1][]{%
    \begingroup%
    \setkeys{badge}{#1}%
        \begin{tikzpicture}[x=1cm,y=1cm, inner sep=0, outer sep=0, anchor=north west]
            % background
            \node (bg) at (0,0) {\includegraphics[width=\badgewidth]{bg_back.png}};

            % all coordinates are relative to background (0,0) to (1,1)
            \begin{scope}[x={(bg.south east)},y={(bg.north west)}]
                % name
                \node[draw=none] at (0,0) {\badge@prename\ \badge@surname};
            \end{scope}
        \end{tikzpicture}%
    \endgroup%
}

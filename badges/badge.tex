\newlength{\badgewidth}

\setlength{\badgeheight}{5cm}

% front
\newcommand*\badgefront[1][]{%
    \begingroup%
    \setkeys{badge}{#1}%
        \definecolor{fontcolor}{HTML}{\badge@fontcolor}%
        \color{fontcolor}%
        \begin{tikzpicture}[x=1cm, y=1cm, inner sep=0, outer sep=0, anchor=north west]
            % background
            \node (bg) at (0,0) {\includegraphics[width=\badgewidth, height=\badgeheight]{\badge@bgfront}};

            % all coordinates are relative to background (0,0) to (1,1)
            \begin{scope}[shift={(bg.north west)}, x={(bg.north)}, y={(bg.west)}, scale=2]
                % name
                \node[draw=none, text width=0.5\badgewidth] at (0.05,0.1) {\Large \badge@prename\ \badge@surname};

                % job
                \node at (0.05,0.5) {\badge@job};
                % role
                \node at (0.05,0.6) {\badge@role};
                % shift
                \node at (0.05,0.8) {\badge@shift};

                % photo
                \ifthenelse{\equal{\badge@photo}{}}
                {
                }
                {
                    \node at (0.6,0.05)  {\includegraphics[width=0.3\badgewidth]{\badge@photo}};
                }
            \end{scope}
        \end{tikzpicture}%
    \endgroup%
}

%back
\newcommand*\badgeback[1][]{%
    \begingroup%
    \setkeys{badge}{#1}%
        \definecolor{fontcolor}{HTML}{\badge@fontcolor}%
        \color{fontcolor}%
        \begin{tikzpicture}[x=1cm,y=1cm, inner sep=0, outer sep=0, anchor=north west]
            % background
            \node (bg) at (0,0) {\includegraphics[width=\badgewidth]{\badge@bgback}};

            % all coordinates are relative to background (0,0) to (1,1)
            \begin{scope}[shift={(bg.north west)}, x={(bg.north)}, y={(bg.west)}, scale=2]
                % rotated name
                \node[rotate=90, text width=\badgeheight, align=center] at (0.02,1) {\small \badge@prename\ \badge@surname};

                % food
                \ifthenelse{\boolean{badge@perm-food}}
                {
                    \node at (0.15,0.35) {Freie Verpflegung};
                }
                {
                    \node at (0.15,0.35) {\sout{Freie Verpflegung}};
                }

                % fiance
                \ifthenelse{\boolean{badge@perm-finance}}
                {
                    \node at (0.15,0.6) {Finanzbereich};
                }
                {
                    \node at (0.15,0.6) {\sout{Finanzbereich}};
                }
            \end{scope}
        \end{tikzpicture}%
    \endgroup%
}
